  template <typename E>
  void reverse(vector<E>& V) {			// reverse a vector
    ArrayStack<E> S(V.size());
    for (int i = 0; i < V.size(); i++)		// push elements onto stack
      S.push(V[i]);
    for (int i = 0; i < V.size(); i++) {	// pop them in reverse order
      V[i] = S.top();  S.pop();
    }
  }
